/*  Left rotate k times   */

#include <math.h>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <assert.h>
#include <limits.h>
#include <stdbool.h>

int main(){
    int n; 
    int k; 
    scanf("%d %d",&n,&k);
    int *a = malloc(sizeof(int) * n);
    for(int a_i = 0; a_i < n; a_i++){
       scanf("%d",&a[(a_i+n-k)%n]);
    }
    
    for(int i = 0;i<n;i++)
        printf("%d ",a[i]);
    return 0;
}

Sampling

Recording an analog signal at evenly spaced instants in time creates samples.


Sampling is the process of recording an analog signal at regular discrete moments of time. The sampling rate fs is the number of samples per second. 
The time interval between samples is called the sampling interval Ts=1/fs.
